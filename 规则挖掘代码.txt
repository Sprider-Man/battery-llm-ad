def load_existing_anomaly_data(data_path):
    """加载数据（移除异常原因列依赖）"""
    try:
        file_ext = os.path.splitext(data_path)[1].lower()
        if file_ext in ['.xlsx', '.xls']:
            df = pd.read_excel(data_path, engine='openpyxl' if file_ext == '.xlsx' else 'xlrd')
        elif file_ext == '.json':
            df = pd.read_json(data_path, orient='records')
        else:
            logger.error(f"不支持的格式：{file_ext}，仅支持.xlsx或.json")
            return None

        # 核心列调整：移除anomaly_reasons，保留状态、电压、异常标记
        # 修改点：使用实际列名
        required_cols = ['整车State状态（状态机编码）', '动力电池内部总电压V1', 'is_anomaly',
                         '动力电池充/放电电流', '1号温度检测点温度']

        # 自动识别单体电压列（列名含"号电池单体电压"）
        mono_voltage_cols = [col for col in df.columns if '号电池单体电压' in col]
        if not mono_voltage_cols:
            logger.error("未找到单体电压列（列名需包含'号电池单体电压'）")
            return None

        logger.info(f"找到单体电压列：{mono_voltage_cols}，共{len(mono_voltage_cols)}个")

        # 检查缺失列
        missing_cols = [col for col in required_cols if col not in df.columns]
        if missing_cols:
            logger.error(f"数据缺少核心列：{missing_cols}")
            return None

        # 过滤有效数据
        df = df[
            (df['is_anomaly'].notna()) &
            (df['动力电池内部总电压V1'].notna()) &
            (df['整车State状态（状态机编码）'].notna()) &
            (df['动力电池充/放电电流'].notna()) &  # 修改点：使用实际列名
            (df['1号温度检测点温度'].notna())  # 修改点：使用实际列名
            ].reset_index(drop=True)

        df['整车State状态（状态机编码）'] = df['整车State状态（状态机编码）'].astype(int)

        # 确保is_anomaly列为布尔类型
        df['is_anomaly'] = df['is_anomaly'].astype(bool)

        logger.info(
            f"成功加载数据：共{len(df)}条，异常样本{df['is_anomaly'].sum()}条，单体电压列{len(mono_voltage_cols)}个")
        logger.info(f"数据列信息: {df[required_cols + mono_voltage_cols[:3]].describe()}")
        return df, mono_voltage_cols
    except Exception as e:
        logger.error(f"加载数据失败：{str(e)}")
        import traceback
        traceback.print_exc()
        return None, None


def calculate_diff_distribution(df, col_name, is_mono=False):
    """计算某列的相邻差值分布（用于分析正常/异常波动）"""
    df_diff = df.copy()
    if is_mono:
        # 单体电压列：按列计算每个单体的相邻差值
        # 注意：这里col_name应该是一个列表
        all_diffs = []
        for col in col_name:
            diffs = df_diff[col].diff().abs().dropna()  # 绝对值，去除首行NaN
            all_diffs.extend(diffs.tolist())
        all_diffs = pd.Series(all_diffs, name='mono_diff')
    else:
        # 非单体列（电流/温度/总电压）：直接计算差值
        all_diffs = df_diff[col_name].diff().abs().dropna()  # 绝对值，去除首行NaN
    return all_diffs


def determine_threshold(normal_diffs, abnormal_diffs, confidence=0.99):
    """基于正常样本分布确定阈值（正常样本中confidence比例的差值小于该值）"""
    if len(normal_diffs) == 0:
        logger.warning("正常样本差值为空，使用默认阈值")
        return 0.1  # 返回一个合理的默认值

    # 移除可能的NaN值
    normal_diffs = normal_diffs.dropna()
    abnormal_diffs = abnormal_diffs.dropna()

    if len(normal_diffs) == 0:
        logger.warning("正常样本差值过滤后为空，使用默认阈值")
        return 0.1

    # 正常样本的分位数作为初始阈值
    threshold = normal_diffs.quantile(confidence)

    # 如果分位数为NaN，使用最大值
    if pd.isna(threshold):
        threshold = normal_diffs.max()
        logger.info(f"分位数为NaN，使用正常样本最大值: {threshold}")

    # 验证阈值：确保异常样本中多数差值超过该阈值
    if len(abnormal_diffs) > 0:
        abnormal_ratio = (abnormal_diffs > threshold).mean()
        if abnormal_ratio < 0.3:  # 如果异常样本中超过阈值的比例低于30%，适当调整
            threshold = np.percentile(
                np.concatenate([normal_diffs.values, abnormal_diffs.values]),
                95  # 取整体95%分位数
            )
            logger.info(f"异常样本超过阈值比例低({abnormal_ratio:.2%})，调整阈值为整体95%分位数: {threshold}")
    else:
        logger.warning("异常样本差值为空，仅基于正常样本计算阈值")

    return round(float(threshold), 3)  # 保留3位小数


def extract_standard_thresholds(df, mono_voltage_cols):
    """从数据分布中提取异常检测阈值（无异常原因列时）"""
    # 分离正常样本和异常样本
    normal_df = df[df['is_anomaly'] == False].copy()
    abnormal_df = df[df['is_anomaly'] == True].copy()

    logger.info(f"正常样本数: {len(normal_df)}, 异常样本数: {len(abnormal_df)}")

    if len(normal_df) == 0 or len(abnormal_df) == 0:
        logger.error("正常样本或异常样本数量为0，无法计算阈值")
        logger.info(f"正常样本数: {len(normal_df)}, 异常样本数: {len(abnormal_df)}")
        return None

    # 分离充电/放电状态（假设状态编码30为放电，20为充电，可根据实际调整）
    discharge_states = [30]  # 放电状态编码
    charge_states = [20]  # 充电状态编码
    discharge_df = df[df['整车State状态（状态机编码）'].isin(discharge_states)]
    charge_df = df[df['整车State状态（状态机编码）'].isin(charge_states)]

    logger.info(f"充电状态样本数: {len(charge_df)}, 放电状态样本数: {len(discharge_df)}")

    thresholds = {}

    # 1. 单体电池电压跳变阈值（共通规则）
    # 计算正常/异常样本的单体电压差值分布
    normal_mono_diffs = calculate_diff_distribution(normal_df, mono_voltage_cols, is_mono=True)
    abnormal_mono_diffs = calculate_diff_distribution(abnormal_df, mono_voltage_cols, is_mono=True)

    if len(normal_mono_diffs) > 0 and len(abnormal_mono_diffs) > 0:
        thresholds['mono_voltage'] = determine_threshold(normal_mono_diffs, abnormal_mono_diffs, confidence=0.99)
        logger.info(
            f"单体电压跳变阈值计算：正常样本{len(normal_mono_diffs)}条，最大{normal_mono_diffs.max():.3f}V，异常样本{len(abnormal_mono_diffs)}条，最小{abnormal_mono_diffs.min():.3f}V，阈值{thresholds['mono_voltage']:.3f}V")
    else:
        thresholds['mono_voltage'] = 0.05  # 默认阈值
        logger.warning(f"单体电压差值数据不足，使用默认阈值: {thresholds['mono_voltage']}V")

    # 2. 温度跳变阈值（共通规则）
    # 修改点：使用实际列名'1号温度检测点温度'
    normal_temp_diffs = calculate_diff_distribution(normal_df, '1号温度检测点温度')
    abnormal_temp_diffs = calculate_diff_distribution(abnormal_df, '1号温度检测点温度')

    if len(normal_temp_diffs) > 0 and len(abnormal_temp_diffs) > 0:
        thresholds['temperature'] = determine_threshold(normal_temp_diffs, abnormal_temp_diffs, confidence=0.99)
        logger.info(
            f"温度跳变阈值计算：正常样本{len(normal_temp_diffs)}条，最大{normal_temp_diffs.max():.1f}℃，异常样本{len(abnormal_temp_diffs)}条，最小{abnormal_temp_diffs.min():.1f}℃，阈值{thresholds['temperature']:.1f}℃")
    else:
        thresholds['temperature'] = 3.0  # 默认阈值
        logger.warning(f"温度差值数据不足，使用默认阈值: {thresholds['temperature']}℃")

    # 3. 电流跳变阈值（充电专属规则）
    if len(charge_df) > 0:
        # 修改点：使用实际列名'动力电池充/放电电流'
        normal_charge_current_diffs = calculate_diff_distribution(charge_df[charge_df['is_anomaly'] == False],
                                                                  '动力电池充/放电电流')
        abnormal_charge_current_diffs = calculate_diff_distribution(charge_df[charge_df['is_anomaly'] == True],
                                                                    '动力电池充/放电电流')

        if len(normal_charge_current_diffs) > 0 and len(abnormal_charge_current_diffs) > 0:
            thresholds['current'] = determine_threshold(normal_charge_current_diffs, abnormal_charge_current_diffs,
                                                        confidence=0.99)
            logger.info(
                f"充电电流跳变阈值计算：正常样本{len(normal_charge_current_diffs)}条，最大{normal_charge_current_diffs.max():.1f}A，异常样本{len(abnormal_charge_current_diffs)}条，最小{abnormal_charge_current_diffs.min():.1f}A，阈值{thresholds['current']:.1f}A")
        else:
            thresholds['current'] = 0.5  # 无充电数据时用默认值
            logger.warning("充电电流差值数据不足，电流跳变阈值使用默认值0.5A")
    else:
        thresholds['current'] = 0.5  # 无充电数据时用默认值
        logger.warning("无充电状态样本，电流跳变阈值使用默认值0.5A")

    # 4. 总电压超限阈值（放电专属规则）
    if len(discharge_df) > 0 and len(discharge_df[discharge_df['is_anomaly'] == False]) > 0:
        # 放电时总电压异常通常为过高，取正常放电总电压的99%分位数作为阈值
        normal_discharge_voltage = discharge_df[discharge_df['is_anomaly'] == False]['动力电池内部总电压V1']
        if len(normal_discharge_voltage) > 0:
            thresholds['discharge_total'] = round(float(normal_discharge_voltage.quantile(0.99)), 1)
            logger.info(
                f"放电总电压阈值计算：正常样本{len(normal_discharge_voltage)}条，最大{normal_discharge_voltage.max():.1f}V，阈值{thresholds['discharge_total']:.1f}V")
        else:
            thresholds['discharge_total'] = 378.2
            logger.warning("无放电正常样本，总电压超限阈值使用默认值378.2V")
    else:
        thresholds['discharge_total'] = 378.2  # 无放电数据时用默认值
        logger.warning("无放电状态样本，总电压超限阈值使用默认值378.2V")

    # 格式转换（添加单位）
    thresholds['mono_voltage'] = f"{thresholds['mono_voltage']:.2f}V"
    thresholds['temperature'] = f"{thresholds['temperature']:.1f}℃"
    thresholds['current'] = f"{thresholds['current']:.1f}A"
    thresholds['discharge_total'] = f"{thresholds['discharge_total']:.1f}V"

    return thresholds


def generate_exact_rule_text(thresholds):
    """生成异常检测规则文本"""
    return f"""# 电池数据异常检测规则

### 1. 充电与放电共通异常规则

| 异常类型 | 判断阈值 | 具体规则 |
|----------|----------|----------|
| 单体电池电压跳变 | {thresholds['mono_voltage']} | 遍历所有'号电池单体电压'列，计算当前行与前一行同一单体电池的电压差值，绝对值超过{thresholds['mono_voltage']}则判定为异常 |
| 温度跳变 | {thresholds['temperature']} | 以1号温度检测点的温度为依据，计算当前行与前一行的温度差值，绝对值超过{thresholds['temperature']}则判定为异常 |

### 2. 充电专属异常规则

| 异常类型 | 判断阈值 | 具体规则 |
|----------|----------|----------|
| 电流跳变 | {thresholds['current']} | 计算当前行与前一行的充/放电电流差值，绝对值超过{thresholds['current']}则判定为异常 |

### 3. 放电专属异常规则

| 异常类型 | 判断阈值 | 具体规则 |
|----------|----------|----------|
| 总电压超限 | {thresholds['discharge_total']} | 直接判断当前行的动力电池内部总电压，若超过{thresholds['discharge_total']}则判定为异常 |"""


def plot_diff_distribution(normal_diffs, abnormal_diffs, title, xlabel, save_path):
    """绘制正常/异常样本的差值分布对比图（辅助验证阈值合理性）"""
    plt.figure(figsize=(10, 6))
    plt.hist(normal_diffs, bins=50, alpha=0.5, label='正常样本', density=True)
    plt.hist(abnormal_diffs, bins=50, alpha=0.5, label='异常样本', density=True)
    if len(normal_diffs) > 0:
        plt.axvline(normal_diffs.quantile(0.99), color='r', linestyle='--', label='阈值（99%分位数）')
    plt.title(title)
    plt.xlabel(xlabel)
    plt.ylabel('概率密度')
    plt.legend()
    plt.savefig(save_path)
    plt.close()
    logger.info(f"分布对比图已保存至：{save_path}")
